CALIFORNIA
══════════════════════════════════════════════════════════════════════════════

-- CA-1. Dataset overview: total reports, date range, unique facilities
SELECT
  COUNT(*) AS total_reports,
  MIN(RECVDATE) AS earliest_report,
  MAX(RECVDATE) AS latest_report,
  COUNT(DISTINCT ASPEN_FACID) AS unique_facilities
FROM
  california;


-- CA-2. Event types ranked by volume + share of total
SELECT
  AE_SHORT,
  AE_CATEGORY,
  COUNT(*) AS n_reports,
  ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER (), 1) AS pct_of_total
FROM
  california
GROUP BY
  AE_SHORT,
  AE_CATEGORY
ORDER BY
  n_reports DESC;


-- CA-3. Substantiated vs. unsubstantiated outcomes
--       FINDING_DETAIL is the accountability column:
--       "Substantiated with deficiencies" = hospital formally cited
SELECT
  FINDING_DETAIL,
  COUNT(*) AS n_reports,
  ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER (), 1) AS pct
FROM
  california
GROUP BY
  FINDING_DETAIL
ORDER BY
  n_reports DESC;


-- CA-4. Hospitals with the most deficiency citations (raw count)
SELECT
  NAME,
  ASPEN_FACTYPE_LABEL,
  COUNT(*) AS cited_incidents
FROM
  california
WHERE
  FINDING_DETAIL = 'Substantiated with deficiencies'
GROUP BY
  NAME,
  ASPEN_FACTYPE_LABEL
ORDER BY
  cited_incidents DESC
LIMIT
  25;


-- CA-5. Hospitals with the highest citation RATE (min 10 reports)
--       Normalises by volume — catches smaller facilities with disproportionate rates
SELECT
  NAME,
  COUNT(*) AS total_reports,
  SUM(
    CASE
      WHEN FINDING_DETAIL = 'Substantiated with deficiencies' THEN 1
      ELSE 0
    END
  ) AS cited,
  ROUND(
    SUM(
      CASE
        WHEN FINDING_DETAIL = 'Substantiated with deficiencies' THEN 1
        ELSE 0
      END
    ) * 100.0 / COUNT(*),
    1
  ) AS citation_rate_pct
FROM
  california
GROUP BY
  NAME
HAVING
  COUNT(*) >= 10
ORDER BY
  citation_rate_pct DESC
LIMIT
  25;


-- CA-6. Criminal events (sexual assault + physical assault) by facility & outcome
SELECT
  NAME,
  AE_SHORT,
  FINDING_DETAIL,
  COUNT(*) AS n_reports
FROM
  california
WHERE
  AE_CATEGORY = '06 - Criminal Events'
GROUP BY
  NAME,
  AE_SHORT,
  FINDING_DETAIL
ORDER BY
  n_reports DESC;


-- CA-7. Outcome breakdown cross-tabbed against event type
--       Shows what share of each incident type gets substantiated
SELECT
  AE_SHORT,
  COUNT(*) AS total,
  SUM(
    CASE
      WHEN FINDING_DETAIL = 'Substantiated with deficiencies' THEN 1
      ELSE 0
    END
  ) AS deficiency_cited,
  SUM(
    CASE
      WHEN FINDING_DETAIL = 'Substantiated without deficiencies' THEN 1
      ELSE 0
    END
  ) AS substantiated_no_deficiency,
  SUM(
    CASE
      WHEN FINDING_DETAIL = 'Unsubstantiated' THEN 1
      ELSE 0
    END
  ) AS unsubstantiated
FROM
  california
GROUP BY
  AE_SHORT
ORDER BY
  total DESC;


-- CA-8. Full incident history for a named hospital (replace name as needed)
SELECT
  RECVDATE,
  AE_SHORT,
  ADVERSE_EVENT,
  FINDING_DETAIL,
  INTAKEID
FROM
  california
WHERE
  NAME LIKE '%GENERAL%' -- replace with target hospital name
ORDER BY
  RECVDATE;

MICHIGAN
══════════════════════════════════════════════════════════════════════════════

-- MI-1. Dataset overview: date range, sentinel event share
SELECT
  COUNT(*) AS total_incidents,
  MIN("Date of Incident") AS earliest,
  MAX("Date of Incident") AS latest,
  SUM("Is this a SUD Sentinel Event?") AS sentinel_events,
  ROUND(
    SUM("Is this a SUD Sentinel Event?") * 100.0 / COUNT(*),
    2
  ) AS sentinel_pct
FROM
  michigan;


-- MI-2. Event types + sub-types ranked by volume
SELECT
  "Event Type",
  "Event Sub-Type",
  COUNT(*) AS n,
  ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER (), 1) AS pct_of_total
FROM
  michigan
GROUP BY
  "Event Type",
  "Event Sub-Type"
ORDER BY
  n DESC;


-- MI-3. All death-related events broken down by sub-type and qualifier
--       Key question: what share of deaths are overdose vs. natural vs. homicide?
SELECT
  "Event Type",
  "Event Sub-Type",
  "Event Sub-Type Qualifier",
  COUNT(*) AS n
FROM
  michigan
WHERE
  "Event Type" IN (
    'Non-Suicide Death',
    'Suicide',
    'Death of Unknown Cause'
  )
GROUP BY
  "Event Type",
  "Event Sub-Type",
  "Event Sub-Type Qualifier"
ORDER BY
  n DESC;


-- MI-4. SUD Sentinel Events only — what types are they?
--       Only 26 of 11,980 incidents flagged — these are the most serious cases
SELECT
  "Event Type",
  "Event Sub-Type",
  "Event Sub-Type Qualifier",
  COUNT(*) AS n
FROM
  michigan
WHERE
  "Is this a SUD Sentinel Event?" = 1
GROUP BY
  "Event Type",
  "Event Sub-Type",
  "Event Sub-Type Qualifier"
ORDER BY
  n DESC;

COLORADO
══════════════════════════════════════════════════════════════════════════════

-- CO-1. Dataset overview: date range, facility count, incidents from closed facilities
SELECT
  COUNT(*) AS total_occurrences,
  MIN("Occurrence Date") AS earliest,
  MAX("Occurrence Date") AS latest,
  COUNT(DISTINCT "Facility ID") AS unique_facilities,
  SUM(
    CASE
      WHEN "Facility Operating Status" = 'Closed' THEN 1
      ELSE 0
    END
  ) AS from_closed_facilities
FROM
  colorado;


-- CO-2. Occurrence types ranked by volume + share of total
SELECT
  "Type of Occurrence",
  COUNT(*) AS n,
  ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER (), 1) AS pct_of_total
FROM
  colorado
GROUP BY
  "Type of Occurrence"
ORDER BY
  n DESC;


-- CO-3. Occurrence types broken down by facility type
--       SNFs vs. assisted living vs. hospitals vs. psychiatric tell different stories
SELECT
  "Facility Type Name",
  "Type of Occurrence",
  COUNT(*) AS n
FROM
  colorado
GROUP BY
  "Facility Type Name",
  "Type of Occurrence"
ORDER BY
  "Facility Type Name",
  n DESC;


-- CO-4. Corporate chains: incidents per facility (abuse & neglect focus)
--       Normalises by facility count to compare chains fairly
SELECT
  "Owner Company",
  COUNT(*) AS n_incidents,
  COUNT(DISTINCT "Facility ID") AS n_facilities,
  ROUND(
    CAST(COUNT(*) AS FLOAT) / COUNT(DISTINCT "Facility ID"),
    1
  ) AS incidents_per_facility
FROM
  colorado
WHERE
  "Type of Occurrence" IN (
    'Physical Abuse',
    'Sexual Abuse',
    'Neglect',
    'Verbal Abuse'
  )
GROUP BY
  "Owner Company"
HAVING
  COUNT(DISTINCT "Facility ID") >= 2
ORDER BY
  incidents_per_facility DESC
LIMIT
  25;


-- CO-5. Geographic breakdown: incidents per county + abuse/neglect subset
SELECT
  "Facility County",
  COUNT(*) AS total_incidents,
  SUM(
    CASE
      WHEN "Type of Occurrence" IN ('Physical Abuse', 'Sexual Abuse', 'Neglect') THEN 1
      ELSE 0
    END
  ) AS abuse_neglect_count,
  COUNT(DISTINCT "Facility ID") AS n_facilities
FROM
  colorado
GROUP BY
  "Facility County"
ORDER BY
  total_incidents DESC;


-- CO-6. Facilities with most serious incidents (abuse + neglect + sexual abuse + brain injury)
--       Includes bed count and owner for facility-profile stories
SELECT
  "Facility Name",
  "Facility County",
  "Owner Company",
  "Bed License Total",
  COUNT(*) AS serious_incidents
FROM
  colorado
WHERE
  "Type of Occurrence" IN (
    'Physical Abuse',
    'Sexual Abuse',
    'Neglect',
    'Brain Injury'
  )
GROUP BY
  "Facility Name",
  "Facility County",
  "Owner Company",
  "Bed License Total"
ORDER BY
  serious_incidents DESC
LIMIT
  25;


-- CO-7. Drug diversion incidents: which facilities, which cities?
SELECT
  "Facility Name",
  "Facility City",
  "Facility County",
  "Owner Company",
  COUNT(*) AS diverted_drug_incidents
FROM
  colorado
WHERE
  "Type of Occurrence" = 'Diverted Drugs'
GROUP BY
  "Facility Name",
  "Facility City",
  "Facility County",
  "Owner Company"
ORDER BY
  diverted_drug_incidents DESC;

WASHINGTON
══════════════════════════════════════════════════════════════════════════════

-- WA-1. Dataset overview (note: SUM the count column, not COUNT(*))
SELECT
  SUM("Adverse Event Count") AS total_incidents,
  MIN(Year) AS earliest_year,
  MAX(Year) AS latest_year,
  COUNT(DISTINCT "Facility Name") AS unique_facilities
FROM
  washington;


-- WA-2. Total incidents per year, 2014–2025 (long-run trend)
--       Washington is the only state with a decade of data — use this for trend stories
SELECT
  Year,
  SUM("Adverse Event Count") AS total_incidents
FROM
  washington
GROUP BY
  Year
ORDER BY
  Year;


-- WA-3. Incident trend by event type across years
--       Replace the WHERE clause with any event type of interest
SELECT
  Year,
  "Event Type",
  SUM("Adverse Event Count") AS n
FROM
  washington
WHERE
  "Event Type" = 'Criminal Events' -- options: Care Management Events, Environmental Events, Procedure Event s,Products or Devices, Patient Protection
GROUP BY
  Year,
  "Event Type"
ORDER BY
  Year;


-- WA-4. All event types ranked for a specific year
SELECT
  "Event Type",
  "Adverse Event",
  SUM("Adverse Event Count") AS n
FROM
  washington
WHERE
  Year = 2024 -- change year as needed
GROUP BY
  "Event Type",
  "Adverse Event"
ORDER BY
  n DESC
LIMIT
  30;


-- WA-5. Facilities with most cumulative incidents (all years) + per-bed rate
--       Per-bed rate normalises for facility size — fairer comparison
SELECT
  "Facility Name",
  "Facility Type",
  Location,
  SUM("Adverse Event Count") AS total_incidents,
  MAX("Facility Size") AS facility_size,
  ROUND(
    SUM("Adverse Event Count") * 1.0 / NULLIF(MAX("Facility Size"), 0),
    2
  ) AS incidents_per_bed
FROM
  washington
GROUP BY
  "Facility Name",
  "Facility Type",
  Location
ORDER BY
  total_incidents DESC
LIMIT
  25;


-- WA-6. Geographic breakdown: total incidents and average per facility by location
SELECT
  Location,
  COUNT(DISTINCT "Facility Name") AS n_facilities,
  SUM("Adverse Event Count") AS total_incidents,
  ROUND(
    SUM("Adverse Event Count") * 1.0 / COUNT(DISTINCT "Facility Name"),
    0
  ) AS avg_incidents_per_facility
FROM
  washington
GROUP BY
  Location
ORDER BY
  total_incidents DESC
LIMIT
  30;


-- WA-7. Facility type breakdown: which types generate the most incidents?
SELECT
  "Facility Type",
  COUNT(DISTINCT "Facility Name") AS n_facilities,
  SUM("Adverse Event Count") AS total_incidents
FROM
  washington
GROUP BY
  "Facility Type"
ORDER BY
  total_incidents DESC;


-- WA-8. Year-over-year change for a specific facility (replace name as needed)
SELECT
  Year,
  SUM("Adverse Event Count") AS total_incidents
FROM
  washington
WHERE
  "Facility Name" LIKE '%HARBORVIEW%' -- replace with target facility name
GROUP BY
  Year
ORDER BY
  Year;

-- CROSS-1. Total incidents per state with date coverage
SELECT
  'California' AS state,
  COUNT(*) AS incidents,
  MIN(RECVDATE) AS earliest,
  MAX(RECVDATE) AS latest
FROM
  california
UNION ALL
SELECT
  'Michigan',
  COUNT(*),
  MIN("Date of Incident"),
  MAX("Date of Incident")
FROM
  michigan
UNION ALL
SELECT
  'Colorado',
  COUNT(*),
  MIN("Occurrence Date"),
  MAX("Occurrence Date")
FROM
  colorado
UNION ALL
SELECT
  'Washington',
  SUM("Adverse Event Count"),
  MIN(CAST(Year AS TEXT)),
  MAX(CAST(Year AS TEXT))
FROM
  washington;



-- CROSS-2. Criminal / abuse incidents compared across CA, CO, WA
--          (MI excluded — no facility names; criminal categories differ by state)
SELECT
  'California' AS state,
  AE_SHORT AS event_type,
  COUNT(*) AS n
FROM
  california
WHERE
  AE_CATEGORY = '06 - Criminal Events'
GROUP BY
  AE_SHORT
UNION ALL
SELECT
  'Colorado',
  "Type of Occurrence",
  COUNT(*)
FROM
  colorado
WHERE
  "Type of Occurrence" IN ('Sexual Abuse', 'Physical Abuse')
GROUP BY
  "Type of Occurrence"
UNION ALL
SELECT
  'Washington',
  "Adverse Event",
  SUM("Adverse Event Count")
FROM
  washington
WHERE
  "Event Type" = 'Criminal Events'
GROUP BY
  "Adverse Event"
ORDER BY
  state,
  n DESC;


-- CROSS-3. Deaths in SUD/behavioral health context — MI vs. CA
--          CA: criminal events + care management; MI: direct death events
SELECT
  'California' AS state,
  AE_SHORT AS event_label,
  COUNT(*) AS n
FROM
  california
WHERE
  AE_CATEGORY IN (
    '06 - Criminal Events',
    '04 - Care Management Events'
  )
GROUP BY
  AE_SHORT
UNION ALL
SELECT
  'Michigan',
  "Event Type" || ' / ' || "Event Sub-Type",
  COUNT(*)
FROM
  michigan
WHERE
  "Event Type" IN (
    'Non-Suicide Death',
    'Suicide',
    'Death of Unknown Cause'
  )
GROUP BY
  "Event Type",
  "Event Sub-Type"
ORDER BY
  state,
  n DESC;
